#include <iostream>

using namespace std;
class pre
{
    int a;
public:

    pre()
    {
        a=0;
    }
    pre(int i)
    {
        a=i;
    }
    pre operator++(int);
    void show();
};

pre pre :: operator++(int)
{
    pre temp;
    int counte=++a;
    temp.a=counte;
    return temp;
}
void pre :: show()
{
    cout<<"a= "<<a;
}
int main()
{
    pre c(100);
    pre c1;
    c++;
    c.show();
    c1.show();

    return 0;
}
